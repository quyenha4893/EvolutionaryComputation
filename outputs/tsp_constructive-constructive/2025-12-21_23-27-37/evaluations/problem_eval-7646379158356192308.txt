def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    if not unvisited_nodes:
        return None

    farthest_node = -1
    max_dist_from_current = -1
    for node in unvisited_nodes:
        dist = distance_matrix[current_node][node]
        if dist > max_dist_from_current:
            max_dist_from_current = dist
            farthest_node = node
    
    if farthest_node == -1:
        return list(unvisited_nodes)[0]

    INITIAL_NODES_ESTIMATE = 30.0
    progress_ratio = 1.0 - (len(unvisited_nodes) / INITIAL_NODES_ESTIMATE)
    alpha = 0.5 + (0.5 * progress_ratio)
    alpha = min(max(alpha, 0.5), 1.0) 

    next_node = None
    best_score = float('inf')

    for candidate_node in unvisited_nodes:
        immediate_cost = distance_matrix[current_node][candidate_node]

        gravity_cost = (distance_matrix[candidate_node][farthest_node] +
                        distance_matrix[candidate_node][destination_node])

        score = (alpha * immediate_cost) + ((1 - alpha) * gravity_cost)

        if score < best_score:
            best_score = score
            next_node = candidate_node
            
    return next_node
