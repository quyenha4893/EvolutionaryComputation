def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    best_node = None
    lowest_score = float('inf')

    if not unvisited_nodes:
        return None

    for candidate_node in unvisited_nodes:
        # Score is the cost to go to the candidate plus the cost from the candidate to the final destination
        score = distance_matrix[current_node][candidate_node] + distance_matrix[candidate_node][destination_node]
