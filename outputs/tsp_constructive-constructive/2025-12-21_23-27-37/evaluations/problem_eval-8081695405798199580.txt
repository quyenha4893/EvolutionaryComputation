def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    next_node = None
    best_score = float('inf')
    
    # This weighting factor, alpha, determines the balance between greedily choosing the
    # nearest node (alpha=1.0) and being pulled towards the destination (alpha=0.0).
    alpha = 0.7  

    if not unvisited_nodes:
        return None

    for candidate_node in unvisited_nodes:
        # Cost to travel from the current node to the candidate
        cost_to_candidate = distance_matrix[current_node][candidate_node]
        
        # Estimated cost from the candidate to the final destination
        cost_from_candidate_to_dest = distance_matrix[candidate_node][destination_node]
        
        # A new evaluation equation is introduced to balance immediate and long-term costs.
        # This score function acts as a heuristic to guide the path selection.
        score = (alpha * cost_to_candidate) + ((1 - alpha) * cost_from_candidate_to_dest)

        if score < best_score:
            best_score = score
            next_node = candidate_node
            
    return next_node
