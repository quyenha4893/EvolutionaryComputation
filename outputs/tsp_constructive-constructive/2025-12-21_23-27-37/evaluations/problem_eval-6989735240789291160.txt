def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    """
    Selects the next node based on a weighted score of the distance from the current node
    and the distance from the candidate node back to the destination.
    """
    next_node = None
    best_score = float('inf')

    if not unvisited_nodes:
        return None

    # Alpha is set to 0.6 to balance the immediate cost (60%) with the cost of
    # returning to the start from the candidate node (40%).
    alpha = 0.6

    for candidate_node in unvisited_nodes:
        # Part 1: Immediate cost from the current node to the candidate.
        dist_to_candidate = distance_matrix[current_node][candidate_node]

        # Part 2: "Future" cost, redefined as the distance from the candidate back to the starting node.
        # This encourages choices that move the path closer to the tour's end.
        dist_to_destination = distance_matrix[candidate_node][destination_node]
        
        # The new score balances the immediate move with closing the loop.
        score = (alpha * dist_to_candidate) + ((1 - alpha) * dist_to_destination)

        if score < best_score:
            best_score = score
            next_node = candidate_node
            
    return next_node
