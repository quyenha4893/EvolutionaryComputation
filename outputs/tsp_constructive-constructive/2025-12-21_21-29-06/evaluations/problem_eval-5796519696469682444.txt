import math

def select_next_node(current_node: int, destination_node: int, unvisited_nodes: set[int], distance_matrix: list[list[float]]) -> int:
    """
    Selects the next node to visit using a heuristic that balances immediate cost with the
    compactness and connectivity of the remaining tour segment.

    Args:
        current_node: The node currently at.
        destination_node: The starting node of the tour, to which the path must return.
        unvisited_nodes: A set of nodes that have not yet been visited.
        distance_matrix: A 2D list (matrix) where distance_matrix[i][j]
                         is the distance between node i and node j.

    Returns:
        The next node to visit.
    """
    if not unvisited_nodes:
        # If all other nodes have been visited, return to the destination node (start node)
        return destination_node

    next_node = -1
    min_total_score = math.inf

    for candidate_node in unvisited_nodes:
        immediate_cost = distance_matrix[current_node][candidate_node]
        
        # Simulate visiting this candidate node by removing it from the unvisited set
        temp_unvisited = unvisited_nodes - {candidate_node}
        
        sum_of_min_remaining_connections = 0.0 

        if not temp_unvisited:
            # If no other nodes are left after selecting candidate_node,
            # the only remaining cost for the future tour segment is the final leg
            # from candidate_node (which will be the last visited node) to the destination.
            sum_of_min_remaining_connections = distance_matrix[candidate_node][destination_node]
        else:
            # Calculate the sum of minimum link costs for each node 'u' in the remaining unvisited set.
            # This represents an estimate of how 'connected' or 'compact' the remaining subproblem is,
            # by summing the shortest path from each remaining node to either another remaining node
            # or the final destination.
            for u in temp_unvisited:
                min_link_for_u = math.inf
                
                # Check direct link from 'u' to the destination node
                min_link_for_u = min(min_link_for_u, distance_matrix[u][destination_node])
                
                # Check links from 'u' to other nodes still in 'temp_unvisited'
                # This loop only runs if there are other unvisited nodes besides 'u' itself.
                if len(temp_unvisited) > 1:
                    for v in temp_unvisited:
                        if u != v:
                            min_link_for_u = min(min_link_for_u, distance_matrix[u][v])
                
                sum_of_min_remaining_connections += min_link_for_u
        
        # The total score combines the immediate cost to reach the candidate node
        # and the estimated future connectivity cost (sum_of_min_remaining_connections),
        # which acts as a proxy for how "easy" or "compact" the rest of the tour will be.
        total_score = immediate_cost + sum_of_min_remaining_connections

        if total_score < min_total_score:
            min_total_score = total_score
            next_node = candidate_node

    return next_node
