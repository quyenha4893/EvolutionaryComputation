import math

def select_next_node(current_node: int, destination_node: int, unvisited_nodes: set[int], distance_matrix: list[list[float]]) -> int:
    """
    Selects the next node to visit using a weighted sum approach, prioritizing
    proximity to the destination node.

    Args:
        current_node: The node currently at.
        destination_node: The starting node of the tour, to which the path must return.
        unvisited_nodes: A set of nodes that have not yet been visited.
        distance_matrix: A 2D list (matrix) where distance_matrix[i][j]
                         is the distance between node i and node j.

    Returns:
        The next node to visit.
    """
    if not unvisited_nodes:
        # If all other nodes have been visited, return to the destination node (start node)
        return destination_node

    next_node = -1
    min_biased_distance = math.inf
    
    # New parameter settings:
    # Assign different weights to direct distance and destination proximity
    weight_direct_distance = 0.4
    weight_to_destination = 0.6  # This gives more weight to getting closer to the destination

    for node in unvisited_nodes:
        # Calculate direct distance from current to candidate node
        direct_distance = distance_matrix[current_node][node]
        # Calculate distance from candidate node to the destination node
        distance_to_destination = distance_matrix[node][destination_node]

        # Combine distances using the new weighted sum equation
        # This aims to select a node that helps reduce the overall path length
        # by considering both the immediate step and its impact on the return leg.
        biased_distance = (direct_distance * weight_direct_distance) + \
                          (distance_to_destination * weight_to_destination)

        if biased_distance < min_biased_distance:
            min_biased_distance = biased_distance
            next_node = node

    return next_node
