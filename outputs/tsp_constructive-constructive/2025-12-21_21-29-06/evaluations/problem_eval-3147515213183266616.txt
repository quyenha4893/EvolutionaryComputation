import math

def select_next_node(current_node: int, destination_node: int, unvisited_nodes: set[int], distance_matrix: list[list[float]]) -> int:
    """
    Selects the next node to visit using a Min-Max Future Connectivity heuristic.
    This heuristic considers both the immediate distance to a candidate node
    and the maximum distance from that candidate to any other remaining
    unvisited nodes or the final destination node. This aims to minimize
    the worst-case 'stretch' for future connections.

    Args:
        current_node: The node currently at.
        destination_node: The starting node of the tour, to which the path must return.
        unvisited_nodes: A set of nodes that have not yet been visited.
        distance_matrix: A 2D list (matrix) where distance_matrix[i][j]
                         is the distance between node i and node j.

    Returns:
        The next node to visit.
    """
    if not unvisited_nodes:
        # If all other nodes have been visited, return to the destination node (start node)
        return destination_node

    next_node = -1
    min_total_score = math.inf

    for candidate_node in unvisited_nodes:
        immediate_cost = distance_matrix[current_node][candidate_node]

        # Determine the "future stretch" or maximum distance from the candidate node
        # to any of the remaining nodes that still need to be visited, including the destination.
        future_stretch_cost = 0.0

        # Create a temporary set of nodes that *would* still need to be visited
        # if 'candidate_node' were chosen next.
        remaining_nodes_after_candidate = unvisited_nodes - {candidate_node}

        if not remaining_nodes_after_candidate:
            # If candidate_node is the last unvisited node, the future stretch is
            # just the distance from candidate_node to the destination.
            future_stretch_cost = distance_matrix[candidate_node][destination_node]
        else:
            # Calculate the maximum distance from the candidate_node to any
            # of the truly remaining unvisited nodes.
            for other_node in remaining_nodes_after_candidate:
                future_stretch_cost = max(future_stretch_cost, distance_matrix[candidate_node][other_node])
            
            # Also consider the distance from the candidate_node to the destination,
            # as that's also a 'future' connection that needs to be made eventually.
            future_stretch_cost = max(future_stretch_cost, distance_matrix[candidate_node][destination_node])
        
        # The total score is the immediate cost plus this calculated future stretch.
        total_score = immediate_cost + future_stretch_cost

        if total_score < min_total_score:
            min_total_score = total_score
            next_node = candidate_node

    return next_node
