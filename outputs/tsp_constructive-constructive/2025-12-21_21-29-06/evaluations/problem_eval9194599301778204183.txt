import math

def select_next_node(current_node: int, destination_node: int, unvisited_nodes: set[int], distance_matrix: list[list[float]]) -> int:
    """
    Selects the next node to visit using a composite score that additively combines
    immediate distance with a dynamically amplified return-to-destination cost.

    Args:
        current_node: The node currently at.
        destination_node: The starting node of the tour, to which the path must return.
        unvisited_nodes: A set of nodes that have not yet been visited.
        distance_matrix: A 2D list (matrix) where distance_matrix[i][j]
                         is the distance between node i and node j.

    Returns:
        The next node to visit.
    """
    if not unvisited_nodes:
        # If all other nodes have been visited, return to the destination node (start node)
        return destination_node

    next_node = -1
    min_total_score = math.inf

    # A small positive weighting factor to dynamically amplify the return cost based on immediate distance.
    # This value helps balance the immediate gain with the long-term impact on tour closure,
    # making the return cost more significant when the immediate step is also large.
    W_boosting_factor = 0.005 

    for candidate_node in unvisited_nodes:
        # 1. Immediate cost to travel from current_node to candidate_node
        immediate_cost = distance_matrix[current_node][candidate_node]
        
        # 2. Direct cost to return to the destination from the candidate_node.
        return_cost_to_destination = distance_matrix[candidate_node][destination_node]
        
        # Combine the costs: immediate cost plus return cost, where the return cost
        # is dynamically boosted by a factor that scales with the immediate cost.
        # This increases the penalty for return_cost_to_destination if the immediate_cost is also high.
        total_score = immediate_cost + return_cost_to_destination * (1 + W_boosting_factor * immediate_cost)

        if total_score < min_total_score:
            min_total_score = total_score
            next_node = candidate_node

    return next_node
