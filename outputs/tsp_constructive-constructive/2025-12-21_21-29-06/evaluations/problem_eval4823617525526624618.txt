import math

def select_next_node(current_node: int, destination_node: int, unvisited_nodes: set[int], distance_matrix: list[list[float]]) -> int:
    """
    Selects the next node to visit based on a modified Nearest Neighbor heuristic
    that considers immediate distance, a one-step look-ahead, and an additional
    subtle bias for candidates that are also close to the destination node.

    Args:
        current_node: The node currently at.
        destination_node: The starting node of the tour, to which the path must return.
        unvisited_nodes: A set of nodes that have not yet been visited.
        distance_matrix: A 2D list (matrix) where distance_matrix[i][j]
                         is the distance between node i and node j.

    Returns:
        The next node to visit.
    """
    if not unvisited_nodes:
        # If all other nodes have been visited, return to the destination node (start node)
        return destination_node

    next_node = -1
    min_combined_cost = math.inf

    # Weighting factor for the one-step look-ahead distance.
    K_lookahead = 0.01 
    # A very small weighting factor for the direct distance from candidate_node to destination_node.
    # This introduces a subtle bias towards candidates that keep the tour path relatively close
    # to the eventual return point, making the final leg potentially cheaper.
    L_return_proximity_bias = 0.005 

    for candidate_node in unvisited_nodes:
        # Step 1: Calculate the immediate cost to move from current_node to candidate_node
        immediate_cost = distance_matrix[current_node][candidate_node]

        # Step 2: Calculate the minimum future distance from candidate_node to another remaining unvisited node.
        # This is a one-step look-ahead from the perspective of the candidate_node.
        min_next_step_cost = math.inf
        remaining_unvisited_after_candidate = unvisited_nodes - {candidate_node}

        if not remaining_unvisited_after_candidate:
            # If candidate_node is the last remaining unvisited node, its next step must be to the destination_node
            min_next_step_cost = distance_matrix[candidate_node][destination_node]
        else:
            # Find the closest node from candidate_node among the *remaining* unvisited nodes
            for future_node in remaining_unvisited_after_candidate:
                future_distance = distance_matrix[candidate_node][future_node]
                if future_distance < min_next_step_cost:
                    min_next_step_cost = future_distance
        
        # Step 3: Calculate the direct distance from the candidate node to the destination node.
        # This term biases towards candidates that are 'well-positioned' for the final return.
        return_to_destination_cost = distance_matrix[candidate_node][destination_node]

        # Combine immediate cost with weighted look-ahead and weighted return-to-destination proximity bias.
        # This new equation balances immediate gain, short-term future prospects, and long-term path closure.
        combined_cost = immediate_cost + K_lookahead * min_next_step_cost + L_return_proximity_bias * return_to_destination_cost

        if combined_cost < min_combined_cost:
            min_combined_cost = combined_cost
            next_node = candidate_node

    return next_node
