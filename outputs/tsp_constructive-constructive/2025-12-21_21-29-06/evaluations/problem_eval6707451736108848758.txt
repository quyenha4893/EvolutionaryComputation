import math

def select_next_node(current_node: int, destination_node: int, unvisited_nodes: set[int], distance_matrix: list[list[float]]) -> int:
    """
    Selects the next node using a combined cost that prioritizes immediate distance
    but applies a dynamic, increasing weight to the foresight component (distance to destination).

    Args:
        current_node: The node currently at.
        destination_node: The starting node of the tour, to which the path must return.
        unvisited_nodes: A set of nodes that have not yet been visited.
        distance_matrix: A 2D list (matrix) where distance_matrix[i][j]
                         is the distance between node i and node j.

    Returns:
        The next node to visit.
    """
    if not unvisited_nodes:
        # If all other nodes have been visited, return to the destination node (start node)
        return destination_node

    next_node = -1
    min_combined_cost = math.inf

    total_nodes = len(distance_matrix)
    
    # Calculate the number of nodes already included in the tour so far.
    # This assumes 'destination_node' is the start and is implicitly part of the tour,
    # and 'current_node' is the last added node.
    num_nodes_in_tour_so_far = total_nodes - len(unvisited_nodes)
    
    # Define parameters for the dynamic foresight weight.
    # 'base_weight' sets the minimum influence of the foresight.
    # 'scaling_factor' determines how much the foresight influence can increase.
    base_weight = 0.05
    scaling_factor = 0.2

    # Calculate the dynamic weight for the foresight component.
    # The weight increases linearly with the progress of the tour (fraction of nodes visited).
    progress_ratio = num_nodes_in_tour_so_far / total_nodes
    dynamic_return_foresight_weight = base_weight + (progress_ratio * scaling_factor)

    for candidate_node in unvisited_nodes:
        # 1. Immediate cost from current_node to candidate_node (primary factor)
        immediate_distance = distance_matrix[current_node][candidate_node]

        # 2. Dynamically weighted distance from the candidate node back to the destination node
        distance_to_destination_from_candidate = distance_matrix[candidate_node][destination_node]

        # Combine the immediate distance with the dynamic-weight foresight component.
        # This biases the choice towards nodes that are not only close but also
        # increasingly help reduce the eventual return distance as the tour progresses.
        combined_cost = immediate_distance + (dynamic_return_foresight_weight * distance_to_destination_from_candidate)

        if combined_cost < min_combined_cost:
            min_combined_cost = combined_cost
            next_node = candidate_node

    return next_node
