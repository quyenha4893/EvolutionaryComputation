import math

def select_next_node(current_node: int, destination_node: int, unvisited_nodes: set[int], distance_matrix: list[list[float]]) -> int:
    """
    Selects the next node to visit using an adaptively biased nearest-neighbor approach.
    The bias towards the destination node increases as more nodes are visited,
    transitioning from a more greedy approach initially to prioritizing closing the loop later.

    Args:
        current_node: The node currently at.
        destination_node: The starting node of the tour, to which the path must return.
        unvisited_nodes: A set of nodes that have not yet been visited.
        distance_matrix: A 2D list (matrix) where distance_matrix[i][j]
                         is the distance between node i and node j.

    Returns:
        The next node to visit.
    """
    if not unvisited_nodes:
        # If all other nodes have been visited, return to the destination node (start node)
        return destination_node

    num_total_nodes = len(distance_matrix)
    
    # initial_nodes_to_visit_count represents the total number of nodes that need to be visited
    # after starting, before returning to the destination.
    # We assume num_total_nodes >= 2 for a meaningful TSP.
    initial_nodes_to_visit_count = num_total_nodes - 1 

    # Calculate the number of nodes already visited in the current path, excluding the starting node.
    nodes_remaining_to_visit = len(unvisited_nodes)
    nodes_visited_so_far_count = initial_nodes_to_visit_count - nodes_remaining_to_visit
    
    # The bias factor increases from 0 (pure nearest neighbor) to a value approaching 1
    # as the tour progresses. This encourages a strong consideration for the destination
    # node's proximity towards the end of the tour.
    bias_factor = nodes_visited_so_far_count / initial_nodes_to_visit_count
    
    next_node = -1
    min_biased_distance = math.inf

    for node in unvisited_nodes:
        direct_distance = distance_matrix[current_node][node]
        distance_to_destination = distance_matrix[node][destination_node]

        # Combine distances with the dynamically adjusted bias factor
        biased_distance = direct_distance + (distance_to_destination * bias_factor)

        if biased_distance < min_biased_distance:
            min_biased_distance = biased_distance
            next_node = node

    return next_node
