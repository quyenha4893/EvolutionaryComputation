import math

def select_next_node(current_node: int, destination_node: int, unvisited_nodes: set[int], distance_matrix: list[list[float]]) -> int:
    """
    Selects the next node to visit using a nearest-neighbor approach biased by
    a combined look-ahead that considers both the next closest unvisited node
    and the proximity to the destination node.

    Args:
        current_node: The node currently at.
        destination_node: The starting node of the tour, to which the path must return.
        unvisited_nodes: A set of nodes that have not yet been visited.
        distance_matrix: A 2D list (matrix) where distance_matrix[i][j]
                         is the distance between node i and node j.

    Returns:
        The next node to visit.
    """
    if not unvisited_nodes:
        # If all other nodes have been visited, return to the destination node (start node)
        return destination_node

    next_node = -1
    min_combined_cost = math.inf
    bias_factor = 0.5  # Weighting factor for the combined look-ahead components

    for candidate_node in unvisited_nodes:
        # Step 1: Calculate the immediate cost to move from current_node to candidate_node
        immediate_cost = distance_matrix[current_node][candidate_node]

        # Step 2: Calculate the minimum future distance from candidate_node to another remaining unvisited node (inspired by No.1)
        # This is a one-step look-ahead from the perspective of the candidate_node.
        min_dist_to_next_unvisited = math.inf
        remaining_unvisited_after_candidate = unvisited_nodes - {candidate_node}

        if not remaining_unvisited_after_candidate:
            # If candidate_node is the last remaining unvisited node, this part becomes the direct path to destination
            min_dist_to_next_unvisited = distance_matrix[candidate_node][destination_node]
        else:
            # Find the closest node from candidate_node among the *remaining* unvisited nodes
            for future_node in remaining_unvisited_after_candidate:
                future_distance = distance_matrix[candidate_node][future_node]
                if future_distance < min_dist_to_next_unvisited:
                    min_dist_to_next_unvisited = future_distance
        
        # Step 3: Calculate the distance from candidate_node to the destination_node (inspired by No.2)
        distance_to_destination = distance_matrix[candidate_node][destination_node]

        # Step 4: Combine these two look-ahead components into a single future cost estimate
        combined_future_component = min_dist_to_next_unvisited + distance_to_destination
        
        # Step 5: Calculate the total combined cost, following No.2's structure
        # immediate_cost + (combined_future_component * bias_factor)
        combined_cost = immediate_cost + (combined_future_component * bias_factor)

        if combined_cost < min_combined_cost:
            min_combined_cost = combined_cost
            next_node = candidate_node

    return next_node
