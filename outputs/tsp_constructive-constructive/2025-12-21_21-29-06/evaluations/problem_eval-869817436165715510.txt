import math

def select_next_node(current_node: int, destination_node: int, unvisited_nodes: set[int], distance_matrix: list[list[float]]) -> int:
    """
    Selects the next node to visit based on a combined heuristic that employs
    a weighted blend of immediate distance and distance to destination,
    further augmented by a look-ahead to the closest other unvisited node.

    Args:
        current_node: The node currently at.
        destination_node: The starting node of the tour, to which the path must return.
        unvisited_nodes: A set of nodes that have not yet been visited.
        distance_matrix: A 2D list (matrix) where distance_matrix[i][j]
                         is the distance between node i and node j.

    Returns:
        The next node to visit.
    """
    if not unvisited_nodes:
        # If all other nodes have been visited, return to the destination node (start node)
        return destination_node

    next_node = -1
    min_combined_cost = math.inf

    # Weighting factor for the distance to destination (inspired by W in Algo 1)
    # This factor (0.2) from Algo 1 is used to balance immediate move and return foresight.
    W_return = 0.2 

    # Weighting factor for the look-ahead to another unvisited node (inspired by W1 in Algo 2)
    W_lookahead = 0.2 

    for candidate_node in unvisited_nodes:
        # 1. Immediate cost from current_node to candidate_node
        immediate_distance = distance_matrix[current_node][candidate_node]

        # 2. Distance from the candidate node back to the destination node
        distance_to_destination_from_candidate = distance_matrix[candidate_node][destination_node]

        # 3. Look-ahead to the closest *other* unvisited node from the candidate
        min_dist_to_other_unvisited = 0.0 # Default if no other unvisited nodes remain
        remaining_after_candidate = unvisited_nodes - {candidate_node}
        
        if remaining_after_candidate:
            min_dist_to_other_unvisited = math.inf
            for other_future_node in remaining_after_candidate:
                min_dist_to_other_unvisited = min(min_dist_to_other_unvisited, distance_matrix[candidate_node][other_future_node])
        
        # Combine components:
        # The primary components (immediate and return-to-destination) are combined
        # using a weighted average inspired by Algo 1.
        # This result is then augmented by the weighted look-ahead component from Algo 2.
        combined_cost = ((1 - W_return) * immediate_distance) + \
                        (W_return * distance_to_destination_from_candidate) + \
                        (W_lookahead * min_dist_to_other_unvisited)

        if combined_cost < min_combined_cost:
            min_combined_cost = combined_cost
            next_node = candidate_node

    return next_node
