import math

def select_next_node(current_node: int, destination_node: int, unvisited_nodes: set[int], distance_matrix: list[list[float]]) -> int:
    """
    Selects the next node to visit using a hybrid approach inspired by both algorithms.
    It considers immediate distance, the closest next unvisited node, and the distance
    to the final destination as part of a weighted look-ahead.

    Args:
        current_node: The node currently at.
        destination_node: The starting node of the tour, to which the path must return.
        unvisited_nodes: A set of nodes that have not yet been visited.
        distance_matrix: A 2D list (matrix) where distance_matrix[i][j]
                         is the distance between node i and node j.

    Returns:
        The next node to visit.
    """
    if not unvisited_nodes:
        # If all other nodes have been visited, return to the destination node (start node)
        return destination_node

    next_node = -1
    min_combined_cost = math.inf
    K = 0.5  # Weighting factor for the combined look-ahead component

    for candidate_node in unvisited_nodes:
        # Step 1: Calculate the immediate cost to move from current_node to candidate_node
        immediate_cost = distance_matrix[current_node][candidate_node]

        # Step 2: Calculate the minimum future distance from candidate_node to another remaining unvisited node
        # This component is 0 if no other unvisited nodes remain after picking the candidate.
        min_dist_to_next_unvisited = 0.0
        remaining_unvisited_after_candidate = unvisited_nodes - {candidate_node}

        if remaining_unvisited_after_candidate:
            # Find the closest node from candidate_node among the *remaining* unvisited nodes
            temp_min_dist = math.inf
            for future_node in remaining_unvisited_after_candidate:
                future_distance = distance_matrix[candidate_node][future_node]
                if future_distance < temp_min_dist:
                    temp_min_dist = future_distance
            min_dist_to_next_unvisited = temp_min_dist
        
        # Step 3: Calculate the distance from candidate_node to the destination_node (always considered)
        distance_to_destination = distance_matrix[candidate_node][destination_node]

        # Step 4: Combine these two look-ahead components into a single future cost estimate, then weight it by K
        # The look-ahead is the sum of the closest connection to another unvisited node (or 0 if none)
        # AND the direct distance back to the destination.
        look_ahead_component = min_dist_to_next_unvisited + distance_to_destination
        
        # Step 5: Calculate the total combined cost, applying K to the integrated look-ahead
        combined_cost = immediate_cost + K * look_ahead_component

        if combined_cost < min_combined_cost:
            min_combined_cost = combined_cost
            next_node = candidate_node

    return next_node
