def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    next_node = None
    best_score = float('inf')

    # Define weights for balancing the two distance factors.
    # A higher weight for the destination encourages a more direct path towards the end.
    weight_current_to_next = 0.4
    weight_next_to_destination = 0.6

    for node in unvisited_nodes:
        # Cost to travel from the current node to this candidate node.
        dist_from_current = distance_matrix[current_node][node]

        # Cost to travel from this candidate node to the final destination.
        dist_to_destination = distance_matrix[node][destination_node]

        # Calculate a composite score. This heuristic avoids picking a node that is
        # close now but very far from the tour's end point.
        score = (weight_current_to_next * dist_from_current) + (weight_next_to_destination * dist_to_destination)

        if score < best_score:
            best_score = score
            next_node = node

    return next_node
