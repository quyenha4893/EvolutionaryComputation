import math

def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    """
    Selects the next node by minimizing a weighted linear score that balances
    the greedy distance against a combined remoteness factor.
    """
    # Algorithm parameters (weights for the linear combination)
    alpha = 1.0  # Weight for the greedy distance component
    beta = 0.5   # Weight for the look-ahead/remoteness bonus

    next_node = None
    best_score = float('inf')

    for candidate_node in unvisited_nodes:
        # Component 1: Greedy distance from the current node to the candidate.
        dist_to_candidate = distance_matrix[current_node][candidate_node]
        
        # Component 2: Look-ahead based on distance to the final destination.
        dist_from_candidate_to_dest = distance_matrix[candidate_node][destination_node]

        # Component 3: Look-ahead based on the candidate's average distance
        # to all other unvisited nodes (its "remoteness").
        other_unvisited_nodes = unvisited_nodes - {candidate_node}
        
        remoteness_factor = 0.0
        if other_unvisited_nodes:
            sum_dist_to_others = sum(distance_matrix[candidate_node][other] for other in other_unvisited_nodes)
            remoteness_factor = sum_dist_to_others / len(other_unvisited_nodes)

        # New Equation: A weighted linear combination. A "bonus" is given for
        # visiting remote nodes (a larger look_ahead_factor reduces the score).
        look_ahead_bonus = dist_from_candidate_to_dest + remoteness_factor
        
        score = alpha * dist_to_candidate - beta * look_ahead_bonus
        
        if score < best_score:
            best_score = score
            next_node = candidate_node
            
    return next_node
