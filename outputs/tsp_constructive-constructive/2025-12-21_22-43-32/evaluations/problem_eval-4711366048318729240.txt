def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    next_node = None
    min_score = float('inf')

    # Weights for balancing the two distance components
    alpha = 0.5  # Weight for the distance from the current node
    beta = 0.5   # Weight for the distance to the destination node

    for node in unvisited_nodes:
        dist_from_current = distance_matrix[current_node][node]
        dist_to_destination = distance_matrix[node][destination_node]
        
        # Calculate a score that balances immediate distance with distance to the end
        score = alpha * dist_from_current + beta * dist_to_destination
        
        if score < min_score:
            min_score = score
            next_node = node
            
    return next_node
