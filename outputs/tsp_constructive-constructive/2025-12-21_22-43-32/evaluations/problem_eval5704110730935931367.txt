def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    """
    Selects the next node by minimizing the ratio of the distance from the current node
    to the distance to the final destination.
    """
    next_node = None
    best_score = float('inf')

    for candidate_node in unvisited_nodes:
        # Cost to travel from the current node to the candidate node
        dist_to_candidate = distance_matrix[current_node][candidate_node]
        
        # Cost to travel from the candidate node to the final destination
        dist_from_candidate_to_dest = distance_matrix[candidate_node][destination_node]

        # The score is a ratio of the two distances. This prioritizes visiting
        # nodes that are close to the current node but relatively far from the
        # destination, effectively pushing the traversal of distant nodes earlier.
        # A small epsilon is added to avoid division by zero.
        score = dist_to_candidate / (dist_from_candidate_to_dest + 1e-9)
        
        if score < best_score:
            best_score = score
            next_node = candidate_node
            
    return next_node
