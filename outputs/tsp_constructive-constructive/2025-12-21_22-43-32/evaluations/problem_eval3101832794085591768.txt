def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    best_node = None
    min_score = float('inf')
    
    # Weighting factor: alpha balances the greediness (distance to next node) 
    # vs. foresight (distance from next node to the tour's end).
    # alpha=1.0 is equivalent to the pure Nearest Neighbor algorithm.
    alpha = 0.7 

    for node in unvisited_nodes:
        # Immediate cost: distance from current node to the candidate node.
        dist_to_next = distance_matrix[current_node][node]
        
        # Future cost estimate: distance from the candidate node back to the start/destination.
        dist_to_dest = distance_matrix[node][destination_node]
        
        # Calculate a blended score to evaluate the candidate node.
        score = (alpha * dist_to_next) + ((1 - alpha) * dist_to_dest)
        
        if score < min_score:
            min_score = score
            best_node = node
            
    return next_node
