def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    next_node = None
    min_detour_cost = float('inf')

    for node in unvisited_nodes:
        # Cost to travel from the current node to this candidate node.
        dist_from_current = distance_matrix[current_node][node]
        # Cost to travel from this candidate node back to the tour's destination.
        dist_to_destination = distance_matrix[node][destination_node]

        # Calculate the total detour cost for inserting this node.
        # This heuristic finds the node that adds the least extra travel
        # distance to the path from the current node to the destination.
        detour_cost = dist_from_current + dist_to_destination

        if detour_cost < min_detour_cost:
            min_detour_cost = detour_cost
            next_node = node
            
    return next_node
