def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    total_nodes = len(distance_matrix)
    if not unvisited_nodes:
        return None

    visited_count = total_nodes - len(unvisited_nodes)
    
    alpha = 0
    if total_nodes > 1:
        alpha = visited_count / (total_nodes - 1)

    best_node = None
    min_weighted_dist = float('inf')

    for node in unvisited_nodes:
        dist_from_current = distance_matrix[current_node][node]
        dist_to_destination = distance_matrix[node][destination_node]
        
        weighted_dist = (1 - alpha) * dist_from_current + alpha * dist_to_destination

        if weighted_dist < min_weighted_dist:
            min_weighted_dist = weighted_dist
            best_node = node
            
    return next_node
